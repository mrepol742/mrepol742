### Code Snippet
```js
// Quirky trick: Swap two variables without a temporary variable using XOR (works only for integers)
function swapXOR(a, b) {
  console.log(`Before swap: a = ${a}, b = ${b}`);
  a = a ^ b;
  b = a ^ b;
  a = a ^ b;
  console.log(`After swap: a = ${a}, b = ${b}`);
  return [a, b];
}

swapXOR(5, 9);
```
